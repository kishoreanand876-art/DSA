class Solution {
    public String shiftingLetters(String s, int[] shifts) {
        int max=0;
       char arr[]=s.toCharArray();
       for(int i=shifts.length-1;i>=0;i--){
           max =(max+ shifts[i]) % 26;
           arr[i]=(char)('a'+(arr[i]-'a'+max)%26);
       }
       return new String(arr);
    }
}
